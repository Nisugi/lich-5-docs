def self.parse(line)
  ALL_DAMAGE.each do |pattern|
    if (match = pattern.match(line))
      result = { damage: match[:damage].to_i }
      result[:target] = match[:target] if match.names.include?('target') && match[:target]
      return result
    end
  end
  nil
end
