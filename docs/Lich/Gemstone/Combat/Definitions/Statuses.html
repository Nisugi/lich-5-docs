def self.parse(line)
  ALL_LOOKUP.each do |pattern, name, action|
    if (match = pattern.match(line))
      result = {
        status: name,
        action: action # :add or :remove
      }
      result[:target] = match[:target] if match.names.include?('target') && match[:target]
      return result
    end
  end
  nil
end
