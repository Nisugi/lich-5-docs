def instance
  @mutex ||= Mutex.new
  @mutex.synchronize {
    @instance ||= begin
      manager = Manager.new

      # Auto-start if enabled in settings
      if manager.settings[:auto_start]
        manager.start
      end

      manager
    end
  }
end
